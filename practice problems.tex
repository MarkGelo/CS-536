				&= wlp(i := 2, wlp(a[j] := a[i] * i, a[i] = 3)) \\ 
				&= wlp(i := 2, i = j ? a[i] * i = 3 : a[i] = 3) \\
				&= j = 2 \; ? \; a[2] * 2 = 3 : a[2] = 3 \\
				&= j = 2 \; ? \; a[2] = \tfrac{3}{2} : a[2] = 3 \\
			\end{aligned}$
			
			\item

				&wlp(x := y, z := x / y, z \geq 9) \\
				&= wlp(x := y, wlp (z := \tfrac{x}{y}, z \geq 9)) \\
				&= wlp(x := y, \tfrac{x}{y} \geq 9) \\
				&= \tfrac{y}{y} \geq 9 \\
				&= 1 \geq 9 \\
				&= F
			\end{aligned}$
